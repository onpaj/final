        Transaction.__table__.update()
        .where(Transaction.id.in_(body.transaction_ids))
        .values(

            categorization_source="manual",
            confidence=None,
        )
    )
    await db.commit()

```python
@router.post("/{batch_id}/retry", response_model=ImportInitiated, status_code=202)
async def retry_batch(
    batch_id: uuid.UUID,
    background_tasks: BackgroundTasks,
    db: AsyncSession = Depends(get_db),
):
    batch = await db.get(ImportBatch, batch_id)
    if not batch or batch.status != "failed":
        raise HTTPException(status_code=400, detail="Only failed batches can be retried")
    batch.status = "processing"
    batch.error_message = None
    await db.commit()
    background_tasks.add_task(_run_import, batch.account_id, b"", batch.filename)
    return ImportInitiated(batch_id=batch.id, message="Retry started")
```

Note: For a real retry, the original file bytes need to be stored or re-uploaded. The simplest M5 approach: store the file in a gitignored local directory (`data/uploads/{batch_id}.csv`) on original upload. Update `_run_import` to read from disk on retry.

- [ ] **Step 1: Create GET /api/settings/llm-cost endpoint**

Create `backend/app/api/settings.py`:

```python
from fastapi import APIRouter, Depends
from sqlalchemy import text
from sqlalchemy.ext.asyncio import AsyncSession
from app.db.session import get_db

router = APIRouter()

@router.get("/llm-cost")
async def llm_cost(db: AsyncSession = Depends(get_db)):
    sql = text("""
        SELECT
            EXTRACT(YEAR FROM created_at)::int AS year,
            EXTRACT(MONTH FROM created_at)::int AS month,
            model,
            COUNT(*) AS calls,
            SUM(prompt_tokens) AS prompt_tokens,
            SUM(completion_tokens) AS completion_tokens
        FROM llm_classifications
        GROUP BY year, month, model
        ORDER BY year DESC, month DESC, model
    """)
    result = await db.execute(sql)
    rows = result.all()
    return [
        {
            "year": r.year, "month": r.month, "model": r.model,
            "calls": r.calls,
            "prompt_tokens": r.prompt_tokens,
            "completion_tokens": r.completion_tokens,
            # Rough cost estimate based on Anthropic pricing (as of 2026-04):
            # Haiku: $0.25/1M input, $1.25/1M output
            # Sonnet: $3/1M input, $15/1M output
            "estimated_cost_usd": round(
                (r.prompt_tokens or 0) / 1_000_000 * (0.25 if "haiku" in r.model else 3.0) +
                (r.completion_tokens or 0) / 1_000_000 * (1.25 if "haiku" in r.model else 15.0),
                4
            ),
        }
        for r in rows
    ]
```

- [ ] **Step 1: Add GET /api/transactions/export endpoint**

```python
import csv
import io
from fastapi.responses import StreamingResponse

@router.get("/export")
async def export_csv(
    account_id: uuid.UUID | None = Query(None),
    date_from: date | None = Query(None),
    date_to: date | None = Query(None),
    db: AsyncSession = Depends(get_db),
):
    q = select(Transaction).order_by(Transaction.booking_date.desc())
    if account_id:
        q = q.where(Transaction.account_id == account_id)
    if date_from:
        q = q.where(Transaction.booking_date >= date_from)
    if date_to:
        q = q.where(Transaction.booking_date <= date_to)
    result = await db.execute(q)
    transactions = result.scalars().all()

    output = io.StringIO()
    writer = csv.writer(output)
    writer.writerow(["date", "amount", "currency", "counterparty", "description",
                     "category_id", "source", "is_transfer"])
    for tx in transactions:
        writer.writerow([
            tx.booking_date, tx.amount, tx.currency,
            tx.counterparty_name or "", tx.description or "",
            tx.category_id or "", tx.categorization_source or "",
            tx.is_transfer,
        ])

    output.seek(0)
    return StreamingResponse(
        iter([output.getvalue()]),
        media_type="text/csv",
        headers={"Content-Disposition": "attachment; filename=transactions.csv"},
    )
```

# Finance Analyzer

Personal local web application for analyzing household finances across multiple bank accounts.

## Prerequisites

- Python 3.11+
- Node.js 20+
- A [Neon](https://neon.tech) Postgres database (free tier)
- An [Anthropic API key](https://console.anthropic.com)

## Setup

### 1. Clone and configure

```bash
git clone <repo-url>
cd FinAl
```

### 2. Backend

```bash
cd backend
python -m venv .venv && source .venv/bin/activate
pip install -e ".[test]"
cp .env.example .env
# Edit .env — set DATABASE_URL and ANTHROPIC_API_KEY
alembic upgrade head
python -m app.db.seed   # load starter categories
```

### 3. Frontend

```bash
cd frontend
npm install
```

## Running

Open two terminals:

**Terminal 1 — Backend:**
```bash
cd backend && source .venv/bin/activate
uvicorn app.main:app --reload
```

**Terminal 2 — Frontend:**
```bash
cd frontend
npm run dev
```

Open **http://localhost:5173** in your browser.

## First Use

1. Go to **Settings** → add your bank account (name + bank type).
2. Go to **Imports** → upload your bank CSV export.
3. Wait for processing to complete (status indicator turns green).
4. Go to **Analytics** → view your spending breakdown.
```
